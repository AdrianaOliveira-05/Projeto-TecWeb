function destinosPossiveis(i, j) {
  const peca = tabuleiroDados[i][j];
  if (!peca || valorDadoAtual === null) return [];

  const steps = valorDadoAtual;
  const player = peca.owner;

  // DFS/BFS de caminhos até "steps"
  const finais = new Set();

  function explore(ci, cj, k) {
    if (k === steps) {
      // não pode terminar numa casa de aliado
      const alvo = tabuleiroDados[ci][cj];
      if (!(alvo && alvo.owner === player)) {
        finais.add(`${ci},${cj}`);
      }
      return;
    }
    const dirs = getCellArrows(ci, cj, player);
    for (const d of dirs) {
      const nxt = stepFrom(ci, cj, d);
      if (!nxt || !dentro(nxt.i, nxt.j)) continue; // caminho inválido
      explore(nxt.i, nxt.j, k + 1);
    }
  }

  explore(i, j, 0);

  // converter para lista [{i,j}]
  const out = [];
  for (const key of finais) {
    const [si, sj] = key.split(",").map(Number);
    out.push({ i: si, j: sj });
  }
  return out;
}
